EAX - Easy Ajax

Ver no GitHub Baixar Demo

Bem vindo à pagina do Easy Ajax.

Essa API foi desenvolvida com o objetivo de facilitar o uso de Ajax em suas aplicações Web. Com isso sua produtividade durante um projeto aumenta e você não precisa se preocupar em criar todo aquele JavaScript.

Com EAX você não precisa mais reinventar a roda, apenas use e começe a produzir.

Como usar

Veja abaixo como fazer uma requisição sem mostrar retorno.

    
// faz um POST
// (String) request, (String) url, (Object) data
// sintaxe: eax.http(req, url, data)

var dados = {
  nome: "Ezequiel"
}

eax.http('POST', '/post-data', dados);

Mostrando uma mensagem de sucesso, erro e progresso.

    
var dados = {
  nome: "Ezequiel"
}

eax
  .http('POST', '/post-data', dados)
  .success(function() {
    alert('Sucesso');
  })
  .error(function() {
    alert('Erro');
  }).
  progress(function() {
    alert('Carregando');
  });

// sintaxe: eax.http(req, url, data).success(function).error(function).error(function);

Envio síncrono e asíncrono.

    
var dados = {
  nome: "Ezequiel"
}

eax
  .http('POST', '/post-data', dados, false) // true: async, false: sync
  .success(function() {
    alert('Sucesso');
  })
  .error(function() {
    alert('Erro');
  }).
  progress(function() {
    alert('Carregando');
  }); 

// sintaxe: eax.http(req, url, data, async).success(function).error(function).error(function);

Recuperando dados

eax
.http('GET', '/get-data', true) // o asincronismo é opcional, caso sem ele, por padrão fica true
.success(function(data) {
  document.body.innerHTML = "Sucesso! Dados: " + data;
})
.error(function(status) {
  alert('Erro! Número do erro: ' + status);
}).
progress(function() {
  alert('Carregando');
});
